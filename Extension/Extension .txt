extension Double{

    

    func sayiyiYuvarla(basamak:Int)->Double{

        let carpan = pow(10.0, Double(basamak))

        return (self*carpan).rounded()/carpan

    }

}

var e = 0.023456

e.sayiyiYuvarla(basamak: 4)

extension Int{

    mutating func kareAl(){

       self = self*self

    }

    

    func kareAlmak()->Int{

        return self*self

    }

    

    func cift()->String{
